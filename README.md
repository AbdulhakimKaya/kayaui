## Install

```bash
npm install --save kayaui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'kayaui'
import 'kayaui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
